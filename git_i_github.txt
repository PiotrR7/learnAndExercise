mkdir mynew project			        -tworzy nowe repozytorium
cd mynewproject/				    -wybór repozytorium
git init					        -inicjalizacja gita
git add <nazwa>					    -dodanie pliku do przechowalni
git rm --cached <nazwa>			    -usuniecie pliku z przechowalni
git commit					        -tworzenie migawki
git status					        -status gita
git show <nazwa> 				    -pokazanie zmian                                                   //wystarczy wpisać tylko 7 znaków
git diff					        -pokazanie ostatnich zmian                                         //wpisując nazwe pliku mozna przegladac zmianny tylko w nim
git checkout <nazwa>				-sprawdzenie commitu, przełączanie między wersjami		           //master - służy do cofniecia sie do ostatnigo zapisanego commitu
git revert <nazwa>				    -cofnięcie commina
git branch <nazwa>				    -tworzenie nowej gałęzi
git branch					        -sprawdzanie dostępnych gałęzi
git branch -d <nazwa>				-usuwanie gałęzi
git merge <nazwa>				    -scalanie gałezi
git push origin <nazwa gałęzi>	    -wypychanie gałęzi do zdalnego repozytorium
git pull origin <nazwa gałęzi>	    -ściąganie gałęzi do lokalnego repozytorium
